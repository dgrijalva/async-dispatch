24/// A handle to a spawned task that can be awaited or detached.
25///
26/// If dropped without calling `detach()`, the task continues running
27/// but its result is discarded.
28#[must_use = "tasks are cancelled when dropped, use .detach() to run in background"]
29pub struct Task<T>(async_task::Task<T>);
30
31impl<T> Task<T> {
32    /// Detach the task, allowing it to run to completion in the background.
33    /// The result will be discarded.
34    pub fn detach(self) {
35        self.0.detach()
36    }
37}
38
39impl<T> Future for Task<T> {
40    type Output = T;
41
42    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
43        Pin::new(&mut self.0).poll(cx)
44    }
45}

47/// Spawn a future on a background GCD queue.
48///
49/// The future will be polled on one of the system's global concurrent queues.
50pub fn spawn<F, T>(future: F) -> Task<T>
51where
52    F: Future<Output = T> + Send + 'static,
53    T: Send + 'static,
54{
55    let (runnable, task) = async_task::spawn(future, schedule_background);
56    runnable.schedule();
57    Task(task)
58}
59
60/// Spawn a future on the main thread's dispatch queue.
61///
62/// Use this for work that must run on the main thread, such as UI updates.
63pub fn spawn_on_main<F, T>(future: F) -> Task<T>
64where
65    F: Future<Output = T> + 'static,
66    T: 'static,
67{
68    let (runnable, task) = async_task::spawn_local(future, schedule_main);
69    runnable.schedule();
70    Task(task)
71}

73/// Spawn a future on a background queue after a delay.
74///
75/// The delay only applies to the initial spawn. If the future yields and is
76/// woken again, subsequent polls happen immediately.
77pub fn spawn_after<F, T>(duration: Duration, future: F) -> Task<T>
78where
79    F: Future<Output = T> + Send + 'static,
80    T: Send + 'static,
81{
82    let first_schedule = Arc::new(AtomicBool::new(true));
83    let (runnable, task) = async_task::spawn(future, move |runnable: async_task::Runnable<()>| {
84        let ptr = runnable.into_raw().as_ptr() as *mut c_void;
85
86        if first_schedule.swap(false, Ordering::SeqCst) {
87            // SAFETY: We call GCD's dispatch_after_f with:
88            // - A valid dispatch_time computed from DISPATCH_TIME_NOW
89            // - A valid global queue handle from dispatch_get_global_queue
90            // - A pointer from Runnable::into_raw() which transfers ownership to GCD
91            // - trampoline, which will reconstruct the Runnable exactly once
92            unsafe {
93                let when = sys::dispatch_time(
94                    sys::DISPATCH_TIME_NOW as u64,
95                    duration.as_nanos() as i64,
96                );
97                sys::dispatch_after_f(
98                    when,
99                    sys::dispatch_get_global_queue(
100                        sys::DISPATCH_QUEUE_PRIORITY_DEFAULT as isize,
101                        0,
102                    ),
103                    ptr,
104                    Some(trampoline),
105                );
106            }
107        } else {
108            // SAFETY: We call GCD's dispatch_async_f with:
109            // - A valid global queue handle from dispatch_get_global_queue
110            // - A pointer from Runnable::into_raw() which transfers ownership to GCD
111            // - trampoline, which will reconstruct the Runnable exactly once
112            unsafe {
113                sys::dispatch_async_f(
114                    sys::dispatch_get_global_queue(
115                        sys::DISPATCH_QUEUE_PRIORITY_DEFAULT as isize,
116                        0,
117                    ),
118                    ptr,
119                    Some(trampoline),
120                );
121            }
122        }
123    });
124    runnable.schedule();
125    Task(task)
126}

128/// Returns a future that completes after the specified duration.
129///
130/// This is the async equivalent of `std::thread::sleep`. The timer is
131/// managed by GCD and does not block any threads while waiting.
132///
133/// Note: The timer cannot be cancelled. If the `Sleep` future is dropped
134/// before completion, the underlying GCD timer still fires but does nothing.
135pub fn sleep(duration: Duration) -> Sleep {
136    Sleep {
137        duration,
138        state: None,
139    }
140}
141
142/// A future that completes after a duration.
143///
144/// Created by the [`sleep`] function.
145pub struct Sleep {
146    duration: Duration,
147    state: Option<Arc<SleepState>>,
148}
149
150struct SleepState {
151    waker: Mutex<Option<Waker>>,
152    completed: AtomicBool,
153}
154
155impl Future for Sleep {
156    type Output = ();
157
158    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<()> {
159        // If we haven't started the timer yet, do so now
160        if self.state.is_none() {
161            let state = Arc::new(SleepState {
162                waker: Mutex::new(Some(cx.waker().clone())),
163                completed: AtomicBool::new(false),
164            });
165
166            // Clone for GCD to own
167            let gcd_state = Arc::clone(&state);
168            let ptr = Arc::into_raw(gcd_state) as *mut c_void;
169
170            // SAFETY: We call GCD's dispatch_after_f with:
171            // - A valid dispatch_time computed from DISPATCH_TIME_NOW
172            // - A valid global queue handle from dispatch_get_global_queue
173            // - A pointer from Arc::into_raw() which transfers one ref count to GCD
174            // - sleep_trampoline, which will call Arc::from_raw() exactly once
175            unsafe {
176                let when = sys::dispatch_time(
177                    sys::DISPATCH_TIME_NOW as u64,
178                    self.duration.as_nanos() as i64,
179                );
180                sys::dispatch_after_f(
181                    when,
182                    sys::dispatch_get_global_queue(
183                        sys::DISPATCH_QUEUE_PRIORITY_DEFAULT as isize,
184                        0,
185                    ),
186                    ptr,
187                    Some(sleep_trampoline),
188                );
189            }
190
191            self.state = Some(state);
192            return Poll::Pending;
193        }
194
195        // Timer already started - check if it's completed
196        let state = self.state.as_ref().unwrap();
197        if state.completed.load(Ordering::SeqCst) {
198            Poll::Ready(())
199        } else {
200            // Update the waker in case it changed
201            *state.waker.lock().unwrap() = Some(cx.waker().clone());
202            Poll::Pending
203        }
204    }
205}

207extern "C" fn sleep_trampoline(context: *mut c_void) {
208    // SAFETY: This pointer was created by Arc::into_raw() in Sleep::poll.
209    // GCD calls this exactly once, so we reclaim the Arc reference here.
210    let state = unsafe { Arc::from_raw(context as *const SleepState) };
211    state.completed.store(true, Ordering::SeqCst);
212    let waker = state.waker.lock().unwrap().take();
213    drop(state);
214    if let Some(waker) = waker {
215        waker.wake();
216    }
217}

219fn dispatch_get_main_queue() -> sys::dispatch_queue_t {
220    std::ptr::addr_of!(sys::_dispatch_main_q) as *const _ as sys::dispatch_queue_t
221}
222
223fn schedule_background(runnable: async_task::Runnable<()>) {
224    let ptr = runnable.into_raw().as_ptr() as *mut c_void;
225    // SAFETY: We call GCD's dispatch_async_f with:
226    // - A valid global queue handle from dispatch_get_global_queue
227    // - A pointer from Runnable::into_raw() which transfers ownership to GCD
228    // - trampoline, which will reconstruct the Runnable exactly once
229    unsafe {
230        sys::dispatch_async_f(
231            sys::dispatch_get_global_queue(sys::DISPATCH_QUEUE_PRIORITY_DEFAULT as isize, 0),
232            ptr,
233            Some(trampoline),
234        );
235    }
236}
237
238fn schedule_main(runnable: async_task::Runnable<()>) {
239    let ptr = runnable.into_raw().as_ptr() as *mut c_void;
240    // SAFETY: We call GCD's dispatch_async_f with:
241    // - The main queue handle (a valid static queue)
242    // - A pointer from Runnable::into_raw() which transfers ownership to GCD
243    // - trampoline, which will reconstruct the Runnable exactly once
244    unsafe {
245        sys::dispatch_async_f(dispatch_get_main_queue(), ptr, Some(trampoline));
246    }
247}
248
249extern "C" fn trampoline(context: *mut c_void) {
250    // SAFETY: This function is only called by GCD with a pointer that was created
251    // by Runnable::into_raw() in one of the schedule functions. GCD guarantees:
252    // - The pointer is passed exactly once per dispatch
253    // - The pointer value is unchanged from what we provided
254    // We reconstruct the Runnable, taking back ownership, and run it.
255    let runnable =
256        unsafe { async_task::Runnable::<()>::from_raw(NonNull::new_unchecked(context as *mut ())) };
257    runnable.run();
258}
